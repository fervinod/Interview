---
- name: Install package and get version
  hosts: linux_hosts
  become: yes
  gather_facts: yes

  vars:
    package_name: nginx

  tasks:

    - name: Install package
      package:
        name: "{{ package_name }}"
        state: present

    - name: Get package version on Debian systems
      shell: "dpkg -s {{ package_name }} | grep Version"
      register: package_version_deb
      when: ansible_os_family == "Debian"

    - name: Get package version on RedHat systems
      shell: "rpm -qi {{ package_name }} | grep Version"
      register: package_version_rhel
      when: ansible_os_family == "RedHat"

    - name: Print package version (Debian)
      debug:
        msg: "{{ inventory_hostname }} -> {{ package_version_deb.stdout }}"
      when: ansible_os_family == "Debian"

    - name: Print package version (RedHat)
      debug:
        msg: "{{ inventory_hostname }} -> {{ package_version_rhel.stdout }}"
      when: ansible_os_family == "RedHat"
