#!/bin/bash

# ==========================================
# RHEL Performance Monitoring Script
# ==========================================

HOSTNAME=$(hostname)
DATE=$(date +"%Y-%m-%d_%H-%M-%S")
LOG_DIR="/var/log/perfmon"
LOG_FILE="${LOG_DIR}/${HOSTNAME}_${DATE}.log"

mkdir -p $LOG_DIR

exec > >(tee -a ${LOG_FILE}) 2>&1

echo "========================================="
echo "RHEL PERFORMANCE REPORT"
echo "Hostname : $HOSTNAME"
echo "Date     : $(date)"
echo "========================================="

# ------------------------------------------
# OS Information
# ------------------------------------------
echo -e "\n[ OS INFORMATION ]"
cat /etc/redhat-release
uname -r

# ------------------------------------------
# Uptime & Load
# ------------------------------------------
echo -e "\n[ SYSTEM LOAD ]"
uptime

# ------------------------------------------
# CPU Statistics
# ------------------------------------------
echo -e "\n[ CPU STATISTICS ]"
mpstat 1 5

# ------------------------------------------
# Memory Statistics
# ------------------------------------------
echo -e "\n[ MEMORY STATISTICS ]"
free -h
vmstat 1 5

# ------------------------------------------
# Disk Usage
# ------------------------------------------
echo -e "\n[ FILESYSTEM USAGE ]"
df -hT

# ------------------------------------------
# Disk I/O Statistics
# ------------------------------------------
echo -e "\n[ DISK I/O ]"
iostat -dxm 1 3

# ------------------------------------------
# Network Statistics
# ------------------------------------------
echo -e "\n[ NETWORK STATISTICS ]"
sar -n DEV 1 3

# ------------------------------------------
# Top CPU Processes
# ------------------------------------------
echo -e "\n[ TOP CPU PROCESSES ]"
ps -eo pid,user,cmd,%mem,%cpu --sort=-%cpu | head -10

# ------------------------------------------
# Top Memory Processes
# ------------------------------------------
echo -e "\n[ TOP MEMORY PROCESSES ]"
ps -eo pid,user,cmd,%mem,%cpu --sort=-%mem | head -10

# ------------------------------------------
# Open TCP Connections
# ------------------------------------------
echo -e "\n[ TCP CONNECTIONS ]"
ss -tunap | head -20

# ------------------------------------------
# Kernel Messages
# ------------------------------------------
echo -e "\n[ RECENT KERNEL MESSAGES ]"
dmesg | tail -20

echo -e "\n========================================="
echo "REPORT SAVED : $LOG_FILE"
echo "========================================="
